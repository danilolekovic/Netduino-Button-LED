Code

var led = new OutputPort(Pins.GPIO_PIN_D13, false);
var button = new InputPort(Pins.GPIO_PIN_D2, false, Port.ResistorMode.Disabled);

while (true)
{
    if (!button.Read())
    {
        led.Write(!led.Read());
        Thread.Sleep(1000);
    }
}
